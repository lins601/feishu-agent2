# 注入组件

# 注入组件


```
import type { IComponentRegistryItem } from '@cvte/mobile-framework';
```


## 文件位置


在 `src/components/business/injectComponents.ts` 注册组件。


## 字段说明


```
interface IComponentRegistryItem {
  code: string;                    // 组件代码（唯一标识）
  title: string;                  // 组件标题
  description: string;             // 组件描述
  editComponent?: ComponentType;  // 编辑组件
  displayComponent?: ComponentType; // 展示组件
  fuzzyComponent?: ComponentType;  // 模糊搜索组件（可选）
  priority?: number;              // 优先级（数字越大优先级越高）
  disableFormItem?: boolean;      // 是否禁用 FormItem 包装
  noRecursive?: boolean;          // 是否不递归处理子元素
}
```


## 注入示例


```
// src/components/business/injectComponents.ts
import type { IComponentRegistryItem } from '@cvte/mobile-framework';
import { baseComponentsConfig, dynamicSSR } from '@cvte/mobile-framework';

export const injectComponents: IComponentRegistryItem[] = [
  ...baseComponentsConfig,  // 基础组件
  {
    code: 'MY_COMPONENT',
    title: '我的组件',
    description: '自定义组件示例',
    editComponent: dynamicSSR(
      () => import('@/components/business/my-component/Edit')
    ),
    displayComponent: dynamicSSR(
      () => import('@/components/business/my-component/Display')
    ),
    priority: 1,
    disableFormItem: false,
    noRecursive: false,
  },
];
```


## 引入和注册


在 `src/components/layout/MobileFrameworkWrapper.tsx` 中引入并注册：


```
// src/components/layout/MobileFrameworkWrapper.tsx
import { registerComponents } from '@cvte/mobile-framework';
import { injectComponents } from '@/components/business/injectComponents';

// 在文件底部注册组件
registerComponents(injectComponents);
```
