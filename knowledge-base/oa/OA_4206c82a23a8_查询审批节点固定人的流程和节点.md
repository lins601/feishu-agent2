# 查询审批节点固定人的流程和节点

```
SELECT
  res.流程名称,
  res.节点名称,
  soe.fd_name AS 姓名,
  res.流程类型,
  res.流程id,
  res.节点id,
  sop.fd_login_name AS 域账号
FROM (
    -- 子查询封装核心逻辑，彻底解决ORA-00904标识符无效问题
    SELECT
      krt.FD_NAME AS 流程名称,
      '原生流程' AS 流程类型,
      krt.FD_ID AS 流程id,
      EXTRACTVALUE(node.COLUMN_VALUE, '/reviewNode/@id') AS 节点id,
      EXTRACTVALUE(node.COLUMN_VALUE, '/reviewNode/@name') AS 节点名称,
      EXTRACTVALUE(node.COLUMN_VALUE, '/reviewNode/@handlerIds') AS 处理人ID
    FROM LBPM_PROCESS_DEFINITION d
    LEFT JOIN LBPM_TEMPLATE lt ON lt.fd_id = d.FD_TEMPLATE_ID
    LEFT JOIN KM_REVIEW_TEMPLATE krt ON krt.fd_id = lt.FD_MODEL_ID,
    TABLE(XMLSEQUENCE(EXTRACT(XMLTYPE(d.fd_content), '/process/nodes/reviewNode'))) node
    WHERE EXTRACTVALUE(node.COLUMN_VALUE, '/reviewNode/@handlerSelectType') = 'org'
      AND d.FD_IS_CURRENT_VERSION = '1'
      AND lt.FD_IS_DELETE = '0'
      AND krt.fd_is_available='1'
      AND krt.fd_id IS NOT NULL
      AND d.fd_content IS NOT NULL
      AND DBMS_LOB.GETLENGTH(d.fd_content) > 0
      AND krt.fd_new_template_url IS null
    UNION ALL
    SELECT
      krt.FD_NAME AS 流程名称,
      '集成流程' AS 流程类型,
      krt.fd_id AS 流程id,
      EXTRACTVALUE(node.COLUMN_VALUE, '/reviewNode/@id') AS 节点id,
      EXTRACTVALUE(node.COLUMN_VALUE, '/reviewNode/@name') AS 节点名称,
      EXTRACTVALUE(node.COLUMN_VALUE, '/reviewNode/@handlerIds') AS 处理人ID
    FROM LBPM_PROCESS_DEFINITION d
    LEFT JOIN LBPM_TEMPLATE lt ON lt.fd_id = d.FD_TEMPLATE_ID
    LEFT JOIN lbpm_docking_template krt ON krt.fd_id = lt.FD_MODEL_ID,
    TABLE(XMLSEQUENCE(EXTRACT(XMLTYPE(d.fd_content), '/process/nodes/reviewNode'))) node
    WHERE EXTRACTVALUE(node.COLUMN_VALUE, '/reviewNode/@handlerSelectType') = 'org'
      AND d.FD_IS_CURRENT_VERSION = '1'
      AND krt.fd_is_available='1'
      AND lt.FD_IS_DELETE = '0'
      AND krt.fd_id IS NOT NULL
      AND d.fd_content IS NOT NULL
      AND DBMS_LOB.GETLENGTH(d.fd_content) > 0
) res
-- 关联人员表获取姓名+域账号，逻辑完全按你提供的SQL整合
LEFT JOIN SYS_ORG_ELEMENT soe ON res.处理人ID = soe.fd_id
LEFT JOIN sys_org_person sop ON soe.fd_id = sop.fd_id
WHERE 1=1 AND soe.fd_org_type = '8'
ORDER BY res.流程id, res.节点id;
```
