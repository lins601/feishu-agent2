# 数据修改删除SQL

```
select *
from (select krm.fd_id,
       krm.doc_subject 标题,
       krm.doc_create_time 申请时间,
       krm.fd_number 流水,
       table_main.fd_applicant 提交人,
       table_main.fd_module 数据所属模块,
       table_main.fd_responsible_dept 数据责任部门,
       table_main.fd_data_area 数据范围说明,
       table_main.fd_reason 申请原因,
       table_main.fd_impact_area 影响范围,
       nvl(table_main.fd_data_system, table_main.fd_system) fd_data_system
from km_review_main krm
     -- 主表
     LEFT JOIN XMLTable('/java/object'
          PASSING XMLTYPE(krm.extend_data_xml)
          COLUMNS fd_applicant VARCHAR2(200) PATH 'void/string[text()="fd_applicant"]/following-sibling::*[1]/void/string[text()="name"]/following-sibling::*[1]/text()'
                 ,fd_data_system VARCHAR2(1024) PATH 'void/string[text()="fd_data_system"]/following-sibling::*[1]/text()'
                 ,fd_responsible_dept VARCHAR2(200) PATH 'void/string[text()="fd_responsible_dept"]/following-sibling::*[1]/void/string[text()="name"]/following-sibling::*[1]/text()'
                 ,fd_module VARCHAR2(200) PATH 'void/string[text()="fd_module"]/following-sibling::*[1]/text()'
                 ,fd_data_area VARCHAR2(200) PATH 'void/string[text()="fd_data_area"]/following-sibling::*[1]/text()'
                 ,fd_reason VARCHAR2(200) PATH 'void/string[text()="fd_reason"]/following-sibling::*[1]/text()'
                 ,fd_impact_area VARCHAR2(200) PATH 'void/string[text()="fd_impact_area"]/following-sibling::*[1]/text()'
                 ,fd_system VARCHAR2(200) PATH 'void/string[text()="fd_system"]/following-sibling::*[1]/text()'
     ) table_main ON 1=1
where krm.fd_template_id = '14f83ba84f088f33824a5044e8c8dc32') t
where (t.fd_data_system like '%CPLM%' or t.fd_data_system like '%Agile%') AND t.申请时间 > TIMESTAMP '2025-07-01 00:02:00.000'
ORDER BY t.申请时间 desc;
```
