# 流程说明

#### 流程说明中内容的配置


 流程owner相关的配置，需要直接在数据库改以下表的字段值


```
-- EKP.EKP_FLOW_OWNER_INFO definition

CREATE TABLE "EKP"."EKP_FLOW_OWNER_INFO"
   (    "TEMPLATE_ID" VARCHAR2(255),
    "TEMPLATE_NAME" VARCHAR2(255),
    "OWNER_DESC" VARCHAR2(2000),
    "OWNER_NAME" VARCHAR2(255),
    "OWNER_ACCOUNT" VARCHAR2(255),
    "IS_SYNC" CHAR(1),
    "BIZ_DOMAIN" VARCHAR2(255),
    "OWNER_POST" VARCHAR2(255),
    "OWNER_DEPART_NAME" VARCHAR2(1000),
    "OWNER_ASSISTANT_ACCOUNT" VARCHAR2(255),
    "OWNER_ASSISTANT_NAME" VARCHAR2(255),
    "APPLY_BG" VARCHAR2(255),
    "APPLY_BU" VARCHAR2(255),
    "REMARK" VARCHAR2(2000),
    "SYSTEM" VARCHAR2(255),
    "CATE_FULL_PATH" VARCHAR2(255),
    "FLOW_FILE" VARCHAR2(2000),
    "IS_KEY_CONTROL" VARCHAR2(255),
    "FLOW_DURATION" VARCHAR2(255)
   )

GRANT SELECT ON "EKP"."EKP_FLOW_OWNER_INFO" TO "XXBI";

COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.BIZ_DOMAIN IS '流程归属域';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.OWNER_POST IS '流程Owner角色';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.OWNER_DEPART_NAME IS '流程Owner部门';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.OWNER_ASSISTANT_ACCOUNT IS '流程代表账号';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.OWNER_ASSISTANT_NAME IS '流程代表姓名';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.APPLY_BG IS '适用BG';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.APPLY_BU IS '适用BU';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.REMARK IS '备注';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.SYSTEM IS '集成系统';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.CATE_FULL_PATH IS '分类全路径';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.FLOW_FILE IS '流程文件/流程文件链接';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.IS_KEY_CONTROL IS '是否重点管控流程';
COMMENT ON COLUMN EKP.EKP_FLOW_OWNER_INFO.FLOW_DURATION IS '流程目标关闭时长（H）（非必填）';
```
