# 查询审批记录

```
SELECT krm.DOC_SUBJECT,n.FD_CREATE_TIME,n.FD_FACT_NODE_NAME ,'https://oa.cvte.com/km/review/km_review_main/kmReviewMain.do?method=view&fdId='||krm.fd_id AS 流程链接
FROM lbpm_audit_note n
LEFT JOIN KM_REVIEW_MAIN krm ON (n.FD_PROCESS_ID =krm.fd_id)
WHERE n.FD_HANDLER_ID  ='16cfddcdfe13b0b78bf56ba43819f249' AND n.FD_CREATE_TIME > TIMESTAMP '2025-11-01 00:00:23.000' AND krm.fd_id IS NOT NULL
UNION ALL
SELECT ldm.DOC_SUBJECT,n.FD_CREATE_TIME,n.FD_FACT_NODE_NAME ,'https://oa.cvte.com/sys/lbpmdocking/lbpm_docking_main/lbpmDockingMain.do?method=view&fdId='||ldm.fd_id AS 流程链接
FROM lbpm_audit_note n
LEFT JOIN LBPM_DOCKING_MAIN ldm   ON (n.FD_PROCESS_ID =ldm.fd_id)
WHERE n.FD_HANDLER_ID  ='16cfddcdfe13b0b78bf56ba43819f249' AND n.FD_CREATE_TIME > TIMESTAMP '2025-11-01 00:00:23.000' AND ldm.fd_id IS NOT null
;
```
